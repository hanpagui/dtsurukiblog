n = int(input('몇 팩토리얼?(!)'))
fct = 1                 # 팩토리얼은 곱하기니까 초기값 0이 아닌 1
for i in range(n,0,-1):
    fct *= i
    print(i,fct)        # fct 곱한값도 같이 보여주기 
print(f'{n}! = {fct}')

예제 > up&down 게임

  1. 1~100 사이의 정답숫자를 랜덤으로 하나 정하고
  2. 정답숫자를 맞출때까지 숫자를 입력

import random
n = random.randint(1,100)
while number1 != n:
    number1=int(input(n))
if number1 > n:
    print('down')
elif number1 < n:
    print('up')
else:
    print(f'정답! {n}')
